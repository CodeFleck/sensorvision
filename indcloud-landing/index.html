Quick Start

Connect your first device in minutes

Our SDKs handle authentication, retries, and connection management so you can focus on your application logic.

quickstart.py 
# Install: pip install indcloud-sdk
from indcloud import IndCloudClient

# Initialize client with your API key
client = IndCloudClient(
    base_url="https://api.indcloud.io",
    api_key="your-api-key"
)

# Send telemetry data
client.send_telemetry(
    device_id="sensor-001",
    variables={
        "temperature": 23.5,
        "humidity": 65.0,
        "pressure": 1013.25
    }
)// Install: npm install indcloud-sdk
import { IndCloudClient } from 'indcloud-sdk';

// Initialize client with your API key
const client = new IndCloudClient({
  baseUrl: 'https://api.indcloud.io',
  apiKey: 'your-api-key'
});

// Send telemetry data
await client.sendTelemetry({
  deviceId: 'sensor-001',
  variables: {
    temperature: 23.5,
    humidity: 65.0,
    pressure: 1013.25
  }
});
